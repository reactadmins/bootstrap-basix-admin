## Examples

<CodePreview>
    <div className="d-flex flex-column gap-2">
        <ProgressBar variant="success" now={25} label="25" striped />
        <ProgressBar variant="info" now={50} label="50" striped />
        <ProgressBar variant="warning" now={75} label="75" striped />
        <ProgressBar variant="danger" now={100} label="100" striped />
    </div>

```jsx title="Progress Bar"
import ProgressBar from "react-bootstrap/ProgressBar";

function MyComponent() {
    return (
        <>
            <ProgressBar variant="success" now={25} label="25" striped />
            <ProgressBar variant="info" now={50} label="50" striped />
            <ProgressBar variant="warning" now={75} label="75" striped />
            <ProgressBar variant="danger" now={100} label="100" striped />
        </>
    );
}

export default MyComponent;
```

### Animated Progress

<CodePreview>
    <div className="d-flex flex-column gap-2">
        <ProgressBar variant="success" now={25} label="25" striped animated />
        <ProgressBar variant="info" now={50} label="50" striped animated />

        <ProgressBar variant="danger" now={100} label="100" striped animated />
    </div>
</CodePreview>

```jsx title="Progress Bar"
import ProgressBar from "react-bootstrap/ProgressBar";

function MyComponent() {
    return (
        <>
            <ProgressBar
                label="25"
                variant="success"
                now={25}
                striped
                Animated
            />
            <ProgressBar 
                label="50"
                variant="info"
                value="50"
                isStriped={true}
                isAnimated={true} />
            <ProgressBar
                label="75"
                variant="warning"
                value="75"
                isStriped={true}
                isAnimated={true}
            />
            <ProgressBar
                label="100"
                variant="danger"
                value="100"
                isStriped={true}
                isAnimated={true}
            />
        </>
    );
}

export default MyComponent;
```

## Props

<FeatureTable>
    <tr>
        <td>variant</td>
        <td>`success`, `danger`, `warning`, `info`</td>
        <td></td>
        <td>Sets the background class of the progress bar.</td>
    </tr>
    <tr>
        <td>now</td>
        <td>number</td>
        <td></td>
        <td>Current value of progress</td>
    </tr>
    <tr>
        <td>label</td>
        <td>node</td>
        <td></td>
        <td>Show label that represents visual percentage. EG. 60%</td>
    </tr>
    <tr>
        <td>striped</td>
        <td>boolean</td>
        <td></td>
        <td>Uses a gradient to create a striped effect.</td>
    </tr>
    <tr>
        <td>animated</td>
        <td>boolean</td>
        <td></td>
        <td>Animate's the stripes from right to left</td>
    </tr>

</FeatureTable>

For more props please visit to <a href="https://react-bootstrap.netlify.app/docs/components/progress/#api" target="_blank">react-bootstrap</a>
